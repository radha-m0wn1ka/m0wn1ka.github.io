temp=(y-1)//2

    spaces=temp-i
    print(" "*spaces,end="")
    pat="* "
    print(pat*(i-1))
for i in range(x-1,-1,-1):
    spaces=temp-i
    print(" "*spaces,end="")
    pat="* "
    print(pat*(i-1))
