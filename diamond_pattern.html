x=10 #max no of stars in line
y=2*x-1 #total line space
temp=(y-1)//2
for i in range(x):
    spaces=temp-i
    print(" "*spaces,end="")
    pat="* "
    print(pat*(i-1))
for i in range(x-1,-1,-1):
    spaces=temp-i
    print(" "*spaces,end="")
    pat="* "
    print(pat*(i-1))
